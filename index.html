import express from 'express'; import cors from 'cors'; import axios from 'axios'; import * as dotenv from 'dotenv'; dotenv.config(); const app = express(); const PORT = process.env.PORT || 8080; app.use(cors()); app.use(express.json()); // النماذج المتاحة فعلياً من المفتاح الجديد const MODELS = [ 'gemini-2.5-flash', 'gemini-2.5-pro', 'gemini-2.0-flash', 'gemini-1.5-flash', 'gemini-1.5-pro', ]; async function callGemini(prompt, apiKey) { for (const model of MODELS) { try { console.log(`🔄 جاري: ${model}`); const res = await axios.post( `https://generativelanguage.googleapis.com/v1/models/${model}:generateContent?key=${apiKey}`, { contents: [{ parts: [{ text: prompt }] }], generationConfig: { maxOutputTokens: 1000, temperature: 0.7 } }, { timeout: 25000 } ); const text = res.data?.candidates?.[0]?.content?.parts?.[0]?.text; if (text) { console.log(`✅ نجح: ${model}`); return { text, model }; } } catch (err) { const status = err.response?.status; const msg = err.response?.data?.error?.message || err.message; console.log(`❌ ${model} → ${status}: ${msg}`); if (status === 400 || status === 403) break; } } throw new Error('فشلت جميع النماذج'); } app.get('/', (req, res) => res.json({ status: '✅ يعمل', version: '3.0.0', api_key_set: !!process.env.GEMINI_API_KEY, endpoints: { test: 'GET /api/test-gemini', extract: 'POST /api/extract-data', generate: 'POST /api/generate-reply' } })); app.get('/health', (req, res) => res.json({ status: 'ok', uptime: process.uptime() })); app.get('/api/test-gemini', async (req, res) => { const apiKey = process.env.GEMINI_API_KEY; if (!apiKey) return res.status(500).json({ success: false, error: 'GEMINI_API_KEY غير موجود' }); try { const result = await callGemini('قل مرحباً بالعربية في جملة واحدة', apiKey); res.json({ success: true, message: '✅ يعمل!', model: result.model, response: result.text }); } catch (err) { res.status(500).json({ success: false, error: err.message }); } }); app.post('/api/extract-data', (req, res) => { try { const { text } = req.body; if (!text) return res.status(400).json({ success: false, error: 'لم يتم توفير نص' }); const get = (regex) => { const m = text.match(regex); return m ? m[1]?.trim() : 'غير محدد'; }; res.json({ success: true, data: { organization: get(/(?:أمانة|وزارة|هيئة|إمارة)\s+([^\n،\.]+)/), refNum: get(/رقم\s*:?\s*([\d]{5,}(?:\/\d)?)/), hijriDate: get(/(14\d{2}\/\d{1,2}\/\d{1,2})/), subject: get(/الموضوع\s*[:/]?\s*(.+?)(?:\n|$)/), position: get(/(?:مدير|رئيس|وكيل|مساعد)\s+(?:عام\s+)?(?:إدارة|قسم|قطاع)?\s*([^\n،\.]{5,40})/) } }); } catch (e) { res.status(500).json({ success: false, error: e.message }); } }); app.post('/api/generate-reply', async (req, res) => { const { data } = req.body; if (!data) return res.status(400).json({ success: false, error: 'لم يتم توفير البيانات' }); const apiKey = process.env.GEMINI_API_KEY; if (!apiKey) return res.status(500).json({ success: false, error: 'GEMINI_API_KEY غير موجود' }); const prompt = `أنت خبير في صياغة الخطابات الإدارية الحكومية السعودية. البيانات: - جهة الصدور: ${data.organization || 'غير محدد'} - المُرسل: ${data.position || 'غير محدد'} - رقم الخطاب: ${data.refNum || 'غير محدد'} - التاريخ: ${data.hijriDate || 'غير محدد'} هـ - الموضوع: ${data.subject || 'غير محدد'} اكتب ديباجة الرد الإداري بهذا الترتيب الدقيق: إشارةً إلى خطاب سعادة ${data.position || '...'} رقم (${data.refNum || '...'}) وتاريخ ${data.hijriDate || '...'}هـ بشأن ${data.subject || '...'}، السلام عليكم ورحمة الله وبركاته، يسعدنا الإحاطة بما يلي: التزم بالعربية الفصحى الرسمية فقط.`; try { const result = await callGemini(prompt, apiKey); res.json({ success: true, reply: result.text, model: result.model }); } catch (err) { res.status(500).json({ success: false, error: err.message }); } }); app.use((err, req, res, next) => res.status(500).json({ success: false, error: err.message })); app.use((req, res) => res.status(404).json({ success: false, error: `غير موجود: ${req.path}` })); app.listen(PORT, () => { console.log(`✅ السيرفر: http://localhost:${PORT}`); console.log(`🔑 API Key: ${process.env.GEMINI_API_KEY ? '✅ موجود' : '❌ مفقود!'}`); }); process.on('SIGTERM', () => process.exit(0));
